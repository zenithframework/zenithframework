### 1. Create ComponentParser.php

**File:** `core/UI/ComponentParser.php`

```php
<?php

declare(strict_types=1);

namespace Zen\UI;

class ComponentParser
{
    public static function parseAttrs(string $raw): array
    {
        $props = [];
        preg_match_all('/([a-zA-Z0-9\-_]+)=["\']([^"\']*)["\']/', $raw, $m, PREG_SET_ORDER);
        
        foreach ($m as $pair) {
            $props[$pair[1]] = $pair[2];
        }
        
        return $props;
    }
    
    public static function parseSlots(string $content): array
    {
        $slots = ['default' => ''];
        
        preg_match_all('/<span\s+slot=["\']([^"\']+)["\']>([\s\S]*?)<\/span>/', $content, $m, PREG_SET_ORDER);
        
        foreach ($m as $slot) {
            $slots[$slot[1]] = $slot[2];
            $content = str_replace($slot[0], '', $content);
        }
        
        $slots['default'] = trim($content);
        
        return $slots;
    }
}
```

### 2. Update Component.php

**File:** `core/UI/Component.php`

Add these methods to the existing class:

```php
use Zen\UI\ComponentParser;

public static function render(string $attrs = '', ?string $slot = null): string
{
    $props = ComponentParser::parseAttrs($attrs);
    $slots = $slot !== null ? ComponentParser::parseSlots($slot) : ['default' => ''];
    
    $instance = new static($props);
    $instance->slots = $slots;
    
    return $instance->render();
}

public function getSlot(string $name = 'default'): string
{
    return $this->slots[$name] ?? '';
}

public function hasSlot(string $name = 'default'): bool
{
    return isset($this->slots[$name]) && $this->slots[$name] !== '';
}
```

### 3. Update TemplateDirectives.php

**File:** `core/UI/TemplateDirectives.php`

Add to `compile()` method:

```php
use App\UI\Components;
use Zen\UI\ComponentParser;

// Zen Components: <zen:button variant="primary" />
protected static function compileZenComponents(string $content): string
{
    // With slot: <zen:card>content</zen:card>
    $content = preg_replace_callback(
        '/<zen:([a-zA-Z0-9\-_]+)\s*([^>]*)>([\s\S]*?)<\/zen:\1>/',
        fn($m) => '<?php echo Components\\' . $m[1] . '::render(\'' . trim($m[2]) . '\', \'' . addslashes($m[3]) . '\'); ?>',
        $content
    );

    // Self-closing: <zen:button variant="primary" />
    $content = preg_replace_callback(
        '/<zen:([a-zA-Z0-9\-_]+)\s+([^>]+)\/>/',
        fn($m) => '<?php echo Components\\' . $m[1] . '::render(\'' . trim($m[2]) . '\'); ?>',
        $content
    );

    // No attributes: <zen:counter />
    $content = preg_replace(
        '/<zen:([a-zA-Z0-9\-_]+)\s*\/>/',
        '<?php echo Components\\$1::render(); ?>',
        $content
    );

    return $content;
}

// Layout: <zen:layout.main title="Home">
protected static function compileLayouts(string $content): string
{
    $content = preg_replace_callback(
        '/<zen:layout\.([a-zA-Z0-9\-_]+)\s*([^>]*)>/',
        fn($m) => self::compileLayoutAttrs($m[1], $m[2]),
        $content
    );
    
    return $content;
}

protected static function compileLayoutAttrs(string $name, string $attrs): string
{
    $props = ComponentParser::parseAttrs($attrs);
    $layout = 'layouts.' . $name;
    
    if (!empty($props)) {
        $sections = [];
        foreach ($props as $key => $value) {
            $sections[] = "ZenTemplate::section('$key'); echo \$value; ZenTemplate::endregion();";
        }
        
        return "<?php ZenTemplate::extends('$layout'); ?>" . 
               implode('', array_map(fn($s) => "<?php $s ?>", $sections));
    }
    
    return "<?php ZenTemplate::extends('$layout'); ?>";
}
```

## 7. Add Optimize:Clear Command

Create new command `php zen optimize:clear` to clear all cache files:

**File:** `core/Console/Commands/OptimizeClear.php`

```php
<?php

declare(strict_types=1);

namespace Zen\Console\Commands;

use Zen\Container;

class OptimizeClear extends Command
{
    protected array $dirs = [
        'storage/framework/views' => 'Compiled views',
        'storage/framework/cache' => 'Application cache',
        'storage/framework/sessions' => 'Session files',
        'storage/logs' => 'Log files',
    ];

    public function handle(Container $container, array $arguments): void
    {
        $cleared = 0;
        $failed = 0;
        
        foreach ($this->dirs as $dir => $label) {
            $path = dirname(__DIR__, 2) . '/' . $dir;
            
            if (!is_dir($path)) {
                continue;
            }
            
            $files = glob($path . '/*');
            foreach ($files as $file) {
                if (is_file($file)) {
                    if (@unlink($file)) {
                        $cleared++;
                    } else {
                        $failed++;
                    }
                }
            }
            
            $this->info("Cleared: {$label}");
        }
        
        $this->info("Total files cleared: {$cleared}");
        
        if ($failed > 0) {
            $this->warn("Failed to clear: {$failed} files");
        }
        
        $this->info("Optimize cache cleared successfully!");
    }
}
```

## 9. Add Development Debug Toolbar (Optional)

Add a floating debug toolbar during development:

- Show execution time
- Memory usage
- Number of queries
- Route matched
- Session data toggle
- View toggle

**File:** `core/Diagnostics/DebugToolbar.php`

```php
<?php

declare(strict_types=1);

namespace Zen\Diagnostics;

class DebugToolbar
{
    protected static float $startTime;
    protected static int $startMemory;
    protected static array $logs = [];
    protected static int $queryCount = 0;
    
    public static function start(): void
    {
        self::$startTime = microtime(true);
        self::$startMemory = memory_get_usage();
    }
    
    public static function recordQuery(string $query): void
    {
        self::$queryCount++;
    }
    
    public static function log(mixed $message): void
    {
        self::$logs[] = [
            'message' => $message,
            'time' => microtime(true) - self::$startTime,
        ];
    }
    
    public static function render(): string
    {
        $time = round((microtime(true) - self::$startTime) * 1000, 2);
        $memory = round((memory_get_usage() - self::$startMemory) / 1024, 2);
        
        return <<<HTML
<div id="debug-toolbar" style="position:fixed;bottom:0;left:0;right:0;background:#1e1e2e;color:#cdd6f4;padding:10px;font-family:monospace;font-size:12px;z-index:99999;display:flex;justify-content:space-between;border-top:2px solid #f38ba8;">
    <div>⏱ {$time}ms | 💾 {$memory}KB | 📊 {$queryCount} queries</div>
    <div>
        <button onclick="document.getElementById('debug-details').style.display=document.getElementById('debug-details').style.display==='none'?'block':'none'">Details</button>
    </div>
</div>
<div id="debug-details" style="display:none;position:fixed;bottom:50px;left:0;right:0;background:#313244;padding:20px;max-height:300px;overflow:auto;border-top:2px solid #89b4fa;">
    <pre style="color:#cdd6f4;">Logs: {$this->formatLogs()}</pre>
</div>
HTML;
    }
    
    protected function formatLogs(): string
    {
        $output = '';
        foreach (self::$logs as $log) {
            $output .= sprintf("[%.2fms] %s\n", $log['time'], $log['message']);
        }
        return $output;
    }
}
```
